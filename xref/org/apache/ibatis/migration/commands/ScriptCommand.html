16  package org.apache.ibatis.migration.commands;

18  import java.io.IOException;
19  import java.io.Reader;
20  import java.math.BigDecimal;
21  import java.util.Collections;
22  import java.util.List;
23  import java.util.Properties;
24  import java.util.StringTokenizer;
25  
26  import org.apache.ibatis.migration.Change;
27  import org.apache.ibatis.migration.MigrationException;
28  import org.apache.ibatis.migration.operations.DatabaseOperation;
29  import org.apache.ibatis.migration.operations.StatusOperation;
30  import org.apache.ibatis.migration.options.SelectedOptions;
31  
32  public final class ScriptCommand extends BaseCommand {
33  
34    public ScriptCommand(SelectedOptions options) {
35      super(options);
36    }
37  
38    @Override
39    public void execute(String... sparams) {
40      try {
41        if (sparams == null || sparams.length < 1 || sparams[0] == null) {
42          throw new MigrationException("The script command requires a range of versions from v1 - v2.");
43        }
44        StringTokenizer parser = new StringTokenizer(sparams[0]);
45        int tokenCount = parser.countTokens();
46        boolean scriptPending = false;
47        boolean scriptPendingUndo = false;
48  
49        String firstToken = parser.nextToken();
50  
51        if (tokenCount == 1 && firstToken.equals("pending")) {
52          scriptPending = true;
53        } else if (tokenCount == 1 && firstToken.equals("pending_undo")) {
54          scriptPendingUndo = true;
55        } else if (!scriptPending && !scriptPendingUndo && tokenCount != 2) {
56          throw new MigrationException("The script command requires a range of versions from v1 - v2.");
57        }
58  
59        BigDecimal v1 = (scriptPending || scriptPendingUndo) ? null : new BigDecimal(firstToken);
60        BigDecimal v2 = (scriptPending || scriptPendingUndo) ? null :new BigDecimal(parser.nextToken());
61  
62        boolean undo;
63        undo = scriptPendingUndo;
64        if(!scriptPending && !scriptPendingUndo) {
65          int comparison = v1.compareTo(v2);
66          if (comparison == 0) {
67            throw new MigrationException("The script command requires two different versions. Use 0 to include the first version.");
68          }
69          undo = comparison > 0;
70        }
71  
72        List<Change> migrations =  (scriptPending || scriptPendingUndo) ?
73                new StatusOperation().operate(getConnectionProvider(), getMigrationLoader(), getDatabaseOperationOption(), null).getCurrentStatus() :
74                getMigrationLoader().getMigrations();
75        Collections.sort(migrations);
76        if (undo) {
77          Collections.reverse(migrations);
78        }
79        for (Change change : migrations) {
80          if (shouldRun(change, v1, v2, scriptPending || scriptPendingUndo)) {
81            printStream.println("-- " + change.getFilename());
82            Reader migrationReader = getMigrationLoader().getScriptReader(change, undo);
83            char[] cbuf = new char[1024];
84            int l;
85            while ((l = migrationReader.read(cbuf)) == cbuf.length) {
86              printStream.print(new String(cbuf, 0, l));
87            }
88  
89            if (l > 0) {
90              printStream.print(new String(cbuf, 0, l - 1));
91            }
92            printStream.println();
93            printStream.println();
94            printStream.println(undo ? generateVersionDelete(change) : generateVersionInsert(change));
95            printStream.println();
96          }
97        }
98      } catch (IOException e) {
99        throw new MigrationException("Error generating script. Cause: " + e, e);
100     }
101   }
102 
103   private String generateVersionInsert(Change change) {
104     return "INSERT INTO " + changelogTable() + " (ID, APPLIED_AT, DESCRIPTION) " +
105         "VALUES (" + change.getId() + ", '" + DatabaseOperation.generateAppliedTimeStampAsString() + "', '"
106         + change.getDescription().replace('\'', ' ') + "')" + getDelimiter();
107   }
108 
109   private String generateVersionDelete(Change change) {
110     return "DELETE FROM " + changelogTable() + " WHERE ID = " + change.getId() + getDelimiter();
111   }
112 
113   private boolean shouldRun(Change change, BigDecimal v1, BigDecimal v2, boolean pendingOnly) {
114     if(!pendingOnly) {
115       BigDecimal id = change.getId();
116       if (v1.compareTo(v2) > 0) {
117         return (id.compareTo(v2) > 0 && id.compareTo(v1) <= 0);
118       } else {
119         return (id.compareTo(v1) > 0 && id.compareTo(v2) <= 0);
120       }
121     }
122     else {
123       return change.getAppliedTimestamp() == null;
124     }
125   }
126 
127   // Issue 699
128   private String getDelimiter() {
129     Properties props = environmentProperties();
130     StringBuilder delimiter = new StringBuilder();
131     if (Boolean.valueOf(props.getProperty("full_line_delimiter"))) delimiter.append("\n"); 
132     delimiter.append(props.getProperty("delimiter", ";"));
133     return delimiter.toString();
134   }
