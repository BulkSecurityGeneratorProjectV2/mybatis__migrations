16  package org.apache.ibatis.migration.operations;
17  
18  import java.io.PrintStream;
19  import java.util.ArrayList;
20  import java.util.Collections;
21  import java.util.List;
22  
23  import org.apache.ibatis.migration.Change;
24  import org.apache.ibatis.migration.ConnectionProvider;
25  import org.apache.ibatis.migration.MigrationLoader;
26  import org.apache.ibatis.migration.options.DatabaseOperationOption;
27  import org.apache.ibatis.migration.utils.Util;
28  
29  public final class StatusOperation extends DatabaseOperation {
30    private int applied;
31  
32    private int pending;
33  
34    private List<Change> changes;
35  
36    public StatusOperation operate(ConnectionProvider connectionProvider, MigrationLoader migrationsLoader,
37        DatabaseOperationOption option, PrintStream printStream) {
38      if (option == null) {
39        option = new DatabaseOperationOption();
40      }
41      println(printStream, "ID             Applied At          Description");
42      println(printStream, Util.horizontalLine("", 80));
43      changes = new ArrayList<Change>();
44      List<Change> migrations = migrationsLoader.getMigrations();
45      if (changelogExists(connectionProvider, option)) {
46        List<Change> changelog = getChangelog(connectionProvider, option);
47        for (Change change : migrations) {
48          int index = changelog.indexOf(change);
49          if (index > -1) {
50            changes.add(changelog.get(index));
51            applied++;
52          } else {
53            changes.add(change);
54            pending++;
55          }
56        }
57      } else {
58        changes.addAll(migrations);
59        pending = migrations.size();
60      }
61      Collections.sort(changes);
62      for (Change change : changes) {
63        println(printStream, change.toString());
64      }
65      println(printStream);
66      return this;
67    }
68  
69    public int getAppliedCount() {
70      return applied;
71    }
72  
73    public int getPendingCount() {
74      return pending;
75    }
76  
77    public List<Change> getCurrentStatus() {
78      return changes;
79    }
80  }
